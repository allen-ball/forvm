021import ball.swing.table.MapTableModel;
022import ball.util.ant.taskdefs.AnnotatedAntTask;
023import ball.util.ant.taskdefs.AntTask;
024import ball.util.ant.taskdefs.ClasspathDelegateAntTask;
025import ball.util.ant.taskdefs.ConfigurableAntTask;
026import ball.util.ant.taskdefs.NotNull;
027import com.vladsch.flexmark.util.ast.Document;
028import forvm.MarkdownService;
029import java.io.File;
030import java.nio.file.Files;
031import lombok.Getter;
032import lombok.NoArgsConstructor;
033import lombok.Setter;
034import lombok.ToString;
035import lombok.experimental.Accessors;
036import org.apache.tools.ant.BuildException;
037import org.apache.tools.ant.Task;
038import org.apache.tools.ant.util.ClasspathUtils;
039
040import static java.nio.charset.StandardCharsets.UTF_8;
041import static lombok.AccessLevel.PROTECTED;
042
043/**
044 * Abstract base class for {@link.uri http://ant.apache.org/ Ant}
045 * {@link Task}s to parse and/or render markdown.
046 *
047 * {@ant.task}
048 *
049 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
050 */
051@NoArgsConstructor(access = PROTECTED)
052public abstract class MarkdownTask extends Task
053                                   implements AnnotatedAntTask,
054                                              ClasspathDelegateAntTask,
055                                              ConfigurableAntTask {
056    @Getter @Setter @Accessors(chain = true, fluent = true)
057    private ClasspathUtils.Delegate delegate = null;
058    protected MarkdownService service = new MarkdownService();
059
060    @Override
061    public void init() throws BuildException {
062        super.init();
063        ClasspathDelegateAntTask.super.init();
064        ConfigurableAntTask.super.init();
065    }
066
067    @Override
068    public void execute() throws BuildException {
069        super.execute();
070        AnnotatedAntTask.super.execute();
071    }
072
073    /**
074     * {@link.uri http://ant.apache.org/ Ant}
075     * {@link org.apache.tools.ant.Task} to parse markdown.
076     *
077     * {@ant.task}
078     */
079    @AntTask("markdown-parse")
080    @NoArgsConstructor @ToString
081    public static class Parse extends MarkdownTask {
082        @NotNull @Getter @Setter
083        private File file = null;
084        protected Document document = null;
085
086        @Override
087        public void execute() throws BuildException {
088            super.execute();
089
090            try {
091                var bytes = Files.readAllBytes(getFile().toPath());
092
093                document = service.parse(new String(bytes, UTF_8));
094
095                var yaml = service.getYamlFrom(document);
096
097                log(new MapTableModel(yaml));
098            } catch (BuildException exception) {
099                throw exception;
100            } catch (Throwable throwable) {
101                throwable.printStackTrace();
102                throw new BuildException(throwable);
103            }
104        }
105    }
106
107    /**
108     * {@link.uri http://ant.apache.org/ Ant}
109     * {@link org.apache.tools.ant.Task} to render markdown to HTML.
110     *
111     * {@ant.task}
112     */
113    @AntTask("markdown-render")
114    @NoArgsConstructor @ToString
115    public static class Render extends Parse {
116        @Override
117        public void execute() throws BuildException {
118            super.execute();
119
120            try {
121                log(service.htmlRender(document, null).toString());
122            } catch (BuildException exception) {
123                throw exception;
124            } catch (Throwable throwable) {
125                throwable.printStackTrace();
126                throw new BuildException(throwable);
127            }
128        }
129    }
130}
