021import forvm.entity.Article;
022import forvm.entity.Author;
023import forvm.repository.ArticleRepository;
024import forvm.repository.AuthorRepository;
025import java.security.Principal;
026import java.util.NoSuchElementException;
027import java.util.Optional;
028import lombok.NoArgsConstructor;
029import lombok.ToString;
030import lombok.extern.log4j.Log4j2;
031import org.apache.commons.io.FilenameUtils;
032import org.springframework.beans.factory.annotation.Autowired;
033import org.springframework.http.HttpMethod;
034import org.springframework.http.HttpStatus;
035import org.springframework.http.ResponseEntity;
036import org.springframework.security.access.AccessDeniedException;
037import org.springframework.security.access.prepost.PreAuthorize;
038import org.springframework.transaction.annotation.Transactional;
039import org.springframework.web.bind.annotation.ExceptionHandler;
040import org.springframework.web.bind.annotation.PathVariable;
041import org.springframework.web.bind.annotation.RequestMapping;
042import org.springframework.web.bind.annotation.RequestParam;
043import org.springframework.web.bind.annotation.ResponseStatus;
044import org.springframework.web.bind.annotation.RestController;
045import org.springframework.web.multipart.MultipartFile;
046
047import static org.springframework.http.MediaType.APPLICATION_JSON_VALUE;
048import static org.springframework.web.bind.annotation.RequestMethod.DELETE;
049import static org.springframework.web.bind.annotation.RequestMethod.POST;
050import static org.springframework.web.bind.annotation.RequestMethod.PUT;
051
052/**
053 * API {@link RestController} implementation.
054 *
055 * {@injected.fields}
056 *
057 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
058 */
059@RestController
060@RequestMapping(value = { "/api/v1" }, produces = APPLICATION_JSON_VALUE)
061@NoArgsConstructor @ToString @Log4j2
062public class APIRestController {
063    @Autowired private AuthorRepository authorRepository = null;
064    @Autowired private ArticleRepository articleRepository = null;
065    @Autowired private MarkdownService service = null;
066
067    @RequestMapping(method = { PUT }, value = { "/author", "/author/{slug}" })
068    @PreAuthorize("hasAuthority('AUTHOR')")
069    @Transactional
070    public ResponseEntity<String> author(Principal principal,
071                                         HttpMethod method,
072                                         @PathVariable Optional<String> slug,
073                                         @RequestParam MultipartFile file) throws Exception {
074        var email = principal.getName();
075        Author author = null;
076
077        switch (method) {
078        case PUT:
079            author = authorRepository.findById(email).get();
080            break;
081
082        default:
083            throw new MethodNotAllowedException(String.valueOf(method));
084        }
085
086        var name = file.getOriginalFilename();
087
088        if (! slug.isPresent()) {
089            slug = Optional.of(FilenameUtils.getBaseName(name));
090        }
091
092        service.compile(name, file.getBytes(), slug.get(), author);
093
094        if (email.equals(author.getEmail())) {
095            authorRepository.save(author);
096        } else {
097            throw new ForbiddenException(slug.get());
098        }
099
100        return new ResponseEntity<>(HttpStatus.OK);
101    }
102
103    @RequestMapping(method = { DELETE }, value = { "/article/{slug}" })
104    @PreAuthorize("hasAuthority('AUTHOR')")
105    @Transactional
106    public ResponseEntity<String> article(Principal principal,
107                                          HttpMethod method,
108                                          @PathVariable String slug) throws Exception {
109        var email = principal.getName();
110
111        switch (method) {
112        case DELETE:
113            var article = articleRepository.findBySlug(slug).get();
114
115            if (email.equals(article.getAuthor().getEmail())) {
116                articleRepository.delete(article);
117            } else {
118                throw new ForbiddenException(slug);
119            }
120            break;
121
122        default:
123            throw new MethodNotAllowedException(String.valueOf(method));
124        }
125
126        return new ResponseEntity<>(HttpStatus.OK);
127    }
128
129    @RequestMapping(method = { POST, PUT },
130                    value = { "/article", "/article/{slug}" })
131    @PreAuthorize("hasAuthority('AUTHOR')")
132    @Transactional
133    public ResponseEntity<String> article(Principal principal,
134                                          HttpMethod method,
135                                          @PathVariable Optional<String> slug,
136                                          @RequestParam MultipartFile file) throws Exception {
137        var email = principal.getName();
138        var name = file.getOriginalFilename();
139
140        if (! slug.isPresent()) {
141            slug = Optional.of(FilenameUtils.getBaseName(name));
142        }
143
144        Article article = null;
145
146        switch (method) {
147        case POST:
148            if (articleRepository.findBySlug(slug.get()).isPresent()) {
149                throw new ConflictException(slug.get());
150            }
151
152            article = new Article();
153            article.setAuthor(authorRepository.findById(email).get());
154            article.setEmail(article.getAuthor().getEmail());
155            break;
156
157        case PUT:
158            article = articleRepository.findBySlug(slug.get()).get();
159            break;
160
161        default:
162            throw new MethodNotAllowedException(String.valueOf(method));
163        }
164
165        service.compile(name, file.getBytes(),
166                        "/article", slug.get(), article);
167
168        if (email.equals(article.getAuthor().getEmail())) {
169            articleRepository.save(article);
170        } else {
171            throw new ForbiddenException(slug.get());
172        }
173
174        return new ResponseEntity<>(HttpStatus.OK);
175    }
176
177    @ExceptionHandler({ NoSuchElementException.class })
178    @ResponseStatus(value = HttpStatus.NOT_FOUND, reason = "Resource not found")
179    public void handleNOT_FOUND() { }
180
181    @ExceptionHandler({ AccessDeniedException.class, SecurityException.class })
182    @ResponseStatus(value = HttpStatus.FORBIDDEN, reason = "Forbidden")
183    public void handleFORBIDDEN() { }
184
185    @ResponseStatus(value = HttpStatus.CONFLICT)
186    public static class ConflictException extends RuntimeException {
187        private static final long serialVersionUID = -3019016947330828859L;
188
189        private ConflictException(String message) { super(message, null); }
190    }
191
192    @ResponseStatus(value = HttpStatus.FORBIDDEN)
193    public static class ForbiddenException extends RuntimeException {
194        private static final long serialVersionUID = -3698565165488957902L;
195
196        private ForbiddenException(String message) { super(message, null); }
197    }
198
199    @ResponseStatus(value = HttpStatus.METHOD_NOT_ALLOWED)
200    public static class MethodNotAllowedException extends RuntimeException {
201        private static final long serialVersionUID = 8201840514907306612L;
202
203        private MethodNotAllowedException(String message) {
204            super(message, null);
205        }
206    }
207}
