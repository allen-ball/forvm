021import forvm.repository.AuthorRepository;
022import forvm.repository.CredentialRepository;
023import forvm.repository.SubscriberRepository;
024import java.util.HashSet;
025import java.util.Set;
026import lombok.NoArgsConstructor;
027import lombok.ToString;
028import lombok.extern.log4j.Log4j2;
029import org.springframework.beans.factory.annotation.Autowired;
030import org.springframework.context.annotation.Bean;
031import org.springframework.context.annotation.Configuration;
032import org.springframework.security.core.GrantedAuthority;
033import org.springframework.security.core.authority.SimpleGrantedAuthority;
034import org.springframework.security.core.userdetails.User;
035import org.springframework.security.core.userdetails.UserDetails;
036import org.springframework.security.core.userdetails.UserDetailsService;
037import org.springframework.security.core.userdetails.UsernameNotFoundException;
038import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserRequest;
039import org.springframework.security.oauth2.client.oidc.userinfo.OidcUserService;
040import org.springframework.security.oauth2.client.userinfo.DefaultOAuth2UserService;
041import org.springframework.security.oauth2.client.userinfo.OAuth2UserRequest;
042import org.springframework.security.oauth2.client.userinfo.OAuth2UserService;
043import org.springframework.security.oauth2.core.OAuth2AuthenticationException;
044import org.springframework.security.oauth2.core.oidc.user.DefaultOidcUser;
045import org.springframework.security.oauth2.core.oidc.user.OidcUser;
046import org.springframework.security.oauth2.core.user.DefaultOAuth2User;
047import org.springframework.security.oauth2.core.user.OAuth2User;
048import org.springframework.transaction.annotation.Transactional;
049
050/**
051 * User details {@link Configuration}.
052 *
053 * {@injected.fields}
054 *
055 * @author {@link.uri mailto:ball@hcf.dev Allen D. Ball}
056 */
057@Configuration
058@NoArgsConstructor @ToString @Log4j2
059public class UserServicesConfiguration {
060    @Autowired private CredentialRepository credentialRepository = null;
061    @Autowired private AuthorRepository authorRepository = null;
062    @Autowired private SubscriberRepository subscriberRepository = null;
063
064
065    @Bean
066    public UserDetailsService userDetailsService() {
067        return new UserDetailsServiceImpl();
068    }
069
070    @Bean
071    public OAuth2UserService<OAuth2UserRequest,OAuth2User> oAuth2UserService() {
072        return new OAuth2UserServiceImpl();
073    }
074
075    @Bean
076    public OidcUserService oidcUserService() {
077        return new OidcUserServiceImpl();
078    }
079
080    private Set<GrantedAuthority> getGrantedAuthoritySet(String username) {
081        var set = new HashSet<GrantedAuthority>();
082
083        subscriberRepository.findById(username)
084            .ifPresent(t -> set.add(new SimpleGrantedAuthority("SUBSCRIBER")));
085
086        authorRepository.findById(username)
087            .ifPresent(t -> set.add(new SimpleGrantedAuthority("AUTHOR")));
088
089        return set;
090    }
091
092    @NoArgsConstructor @ToString
093    private class UserDetailsServiceImpl implements UserDetailsService {
094        @Override
095        @Transactional(readOnly = true)
096        public UserDetails loadUserByUsername(String username) throws UsernameNotFoundException {
097            User user = null;
098
099            try {
100                var credential = credentialRepository.findById(username);
101
102                user = new User(username, credential.get().getPassword(), getGrantedAuthoritySet(username));
103            } catch (UsernameNotFoundException exception) {
104                throw exception;
105            } catch (Exception exception) {
106                throw new UsernameNotFoundException(username);
107            }
108
109            return user;
110        }
111    }
112
113    @NoArgsConstructor @ToString
114    private class OAuth2UserServiceImpl extends DefaultOAuth2UserService {
115        private final DefaultOAuth2UserService delegate = new DefaultOAuth2UserService();
116
117        @Override
118        @Transactional(readOnly = true)
119        public OAuth2User loadUser(OAuth2UserRequest request) throws OAuth2AuthenticationException {
120            var attribute =
121                request.getClientRegistration().getProviderDetails()
122                .getUserInfoEndpoint().getUserNameAttributeName();
123            var user = delegate.loadUser(request);
124
125            try {
126                user = new DefaultOAuth2User(getGrantedAuthoritySet(user.getName()), user.getAttributes(), attribute);
127            } catch (OAuth2AuthenticationException exception) {
128                throw exception;
129            } catch (Exception exception) {
130                log.warn("{}", request, exception);
131            }
132
133            return user;
134        }
135    }
136
137    @NoArgsConstructor @ToString
138    private class OidcUserServiceImpl extends OidcUserService {
139        { setOauth2UserService(oAuth2UserService()); }
140
141        @Override
142        @Transactional(readOnly = true)
143        public OidcUser loadUser(OidcUserRequest request) throws OAuth2AuthenticationException {
144            var attribute =
145                request.getClientRegistration().getProviderDetails()
146                .getUserInfoEndpoint().getUserNameAttributeName();
147            var user = super.loadUser(request);
148
149            try {
150                user =
151                    new DefaultOidcUser(getGrantedAuthoritySet(user.getName()),
152                                        user.getIdToken(), user.getUserInfo(), attribute);
153            } catch (OAuth2AuthenticationException exception) {
154                throw exception;
155            } catch (Exception exception) {
156                log.warn("{}", request, exception);
157            }
158
159            return user;
160        }
161    }
162}
